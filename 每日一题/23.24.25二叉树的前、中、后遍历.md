![](https://img-blog.csdnimg.cn/img_convert/a39a9d0b50a5993b65495092d53c0200.png)

每次递归，相当于对不同目标进行函数体内容操作。

## 2.可以考虑栈

以前序为例

根节点入栈，

当栈非空时，进行出栈并加入列表，并且判断右左孩子节点是否为空，非空入栈（前序先访问左节点，入栈时应该后入）

```java
import java.util.*;

/*
 * public class TreeNode {
 *   int val = 0;
 *   TreeNode left = null;
 *   TreeNode right = null;
 *   public TreeNode(int val) {
 *     this.val = val;
 *   }
 * }
 */

public class Solution {
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param root TreeNode类 
     * @return int整型一维数组
     */
    public int[] preorderTraversal (TreeNode root) {
        // write code here   左中右 栈
        if(root == null) return new int[0];
        Stack<TreeNode> stack = new Stack<>();
        stack.push(root);
        List<Integer> list = new ArrayList<>();
        TreeNode t;
        while(!stack.isEmpty()){
            t = stack.pop();
            list.add(t.val);
            if(t.right != null){
                stack.push(t.right);
            }
            if(t.left != null){
                stack.push(t.left);
            }
        }
        int[] a = new int[list.size()];
        for(int i = 0;i < a.length;i++){
            a[i] = list.get(i);
        }
        return a;
    }
}
```

```java
 *   TreeNode right = null;

 *     this.val = val;

    public int[] preorderTraversal (TreeNode root) {
        // write code here   左中右 栈
        List<Integer> list = new ArrayList<>();
        pre(list,root);
        int[] a = new int[list.size()];
        for(int i = 0;i < a.length;i++){
            a[i] = list.get(i);
        }
        return a;
    }
    public void pre(List<Integer> list,TreeNode root){
        if(root != null){
            list.add(root.val);
            pre(list,root.left);   //根左右
            pre(list,root.right);
        }
    }
}
```

## 后序：

```java
import java.util.*;

/*
 * public class TreeNode {
 *   int val = 0;
 *   TreeNode left = null;
 *   TreeNode right = null;
 *   public TreeNode(int val) {
 *     this.val = val;
 *   }
 * }
 */

public class Solution {
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param root TreeNode类 
     * @return int整型一维数组
     */
    public int[] postorderTraversal (TreeNode root) {
        // write code here
        List<Integer> list = new ArrayList<>();
        end(list,root);
        int[] a = new int[list.size()];
        for(int i = 0;i < a.length;i++) a[i] = list.get(i);
        return a;
    }
    public void end(List<Integer> list,TreeNode root){
        if(root != null){
            end(list,root.left);
            end(list,root.right);
            list.add(root.val);
        }
    }
}
```
